# ToggleButton

ToggleButton with single selected value

<Preview>
  <Story name="ToggleButton1">
  {
    () => {
        const [value, setValue] = useState('Middle');
        const items = ['Left', 'Middle', 'Right'].map((value) => ({ value }));
        return (
          <ToggleButton items={items} value={value} onChange={(event, val) => setValue(val)} />
        )
      }
  }
  </Story>
</Preview>
